create table Students(
   id int NOT NULL,
    zalik int not null unique,
    student_name char(60) not null,
    age int,
    facult_id int not null,
    constraint fk_facult_id
                     foreign key (facult_id) references Facultety(id)
                     on delete restrict
                     on update restrict
   );

create table Facultety(
    id int not null primary key,
    name_facultetu char(120) not  null
);

create table Subjects(
  id int not null primary key,
  name_subject char(60) not null
);

create table Marks(
   id int not null  primary key auto_increment,
   stdnt_id int not null,
    constraint fk_stdnt_id
                  foreign key (stdnt_id) references Students(zalik)
                  on delete restrict
                  on update restrict,
    subj_id int not null,
    constraint fk_subj_id
                  foreign key (subj_id) references subjects(id)
                  on delete restrict
                  on update restrict
);
